def generate_crawl_summary(crawl_reports):
    """Generate a concise summary from multiple crawl reports"""
    summary = "ZEEGUU CRAWLER SUMMARY (Round-Robin)\n"
    summary += "=" * 60 + "\n\n"

    total_new = 0
    total_in_db = 0
    total_low_quality = 0
    total_errors = 0
    total_unprocessed = 0
    total_readability_timeout = 0

    for lang_code, report in crawl_reports.items():
        lang_data = report.data["lang"][lang_code]
        lang_new = 0
        lang_in_db = 0
        lang_low_quality = 0
        lang_errors = 0
        lang_unprocessed = 0
        lang_readability_timeout = 0

        summary += f"\n{lang_code.upper()}\n"
        summary += "-" * 40 + "\n"

        for feed_id, feed_data in lang_data["feeds"].items():
            feed = Feed.find_by_id(feed_id)
            if not feed:
                continue

            new = feed_data.get("total_downloaded", 0) or 0
            in_db = feed_data.get("total_in_db", 0) or 0
            low_q = feed_data.get("total_low_quality", 0) or 0
            errors = len(feed_data.get("feed_errors", []))
            unprocessed = feed_data.get("total_skipped_unprocessed", 0) or 0
            readability_timeout = feed_data.get("total_readability_timeout", 0) or 0

            lang_new += new
            lang_in_db += in_db
            lang_low_quality += low_q
            lang_errors += errors
            lang_unprocessed += unprocessed
            lang_readability_timeout += readability_timeout

            # Only show feeds with activity, errors, unprocessed articles, or readability timeouts
            if new > 0 or errors > 0 or unprocessed > 0 or readability_timeout > 0:
                status = ""
                if errors > 0:
                    status = f" [ERROR]"
                elif new == 0 and in_db > 0:
                    status = " [all in DB]"

                feed_time = feed_data.get("crawl_time", 0) or 0
                # Mark feeds that hit the 5-minute limit
                if feed_time >= 300:
                    status += " [MAX_TIME_REACHED]"

                unproc_str = f" | Unproc: {unprocessed:3d}" if unprocessed > 0 else ""
                rdbl_str = f" | RdblTO: {readability_timeout:3d}" if readability_timeout > 0 else ""
                summary += f"  {feed.title[:40]:40s} | New: {new:3d} | InDB: {in_db:3d} | LowQ: {low_q:3d}{unproc_str}{rdbl_str} | Time: {feed_time:5.1f}s{status}\n"

        total_new += lang_new
        total_in_db += lang_in_db
        total_low_quality += lang_low_quality
        total_errors += lang_errors
        total_unprocessed += lang_unprocessed
        total_readability_timeout += lang_readability_timeout

        summary += f"\n  {lang_code.upper()} Total: {lang_new} new, {lang_in_db} already in DB, {lang_low_quality} low quality"
        if lang_unprocessed > 0:
            summary += f", {lang_unprocessed} unprocessed"
        if lang_readability_timeout > 0:
            summary += f", {lang_readability_timeout} readability timeouts"
        if lang_errors > 0:
            summary += f", {lang_errors} errors"
        summary += f" | Time: {lang_data.get('total_time', 0) or 0:.1f}s\n"

    # Calculate total time
    total_time = sum(
        report.data["lang"][lang_code].get("total_time", 0) or 0
        for lang_code, report in crawl_reports.items()
    )

    summary += "\n" + "=" * 60 + "\n"
    summary += f"OVERALL: {total_new} new articles, {total_in_db} already in DB, {total_low_quality} low quality"
    if total_unprocessed > 0:
        summary += f", {total_unprocessed} unprocessed"
    if total_readability_timeout > 0:
        summary += f", {total_readability_timeout} readability timeouts"
    if total_errors > 0:
        summary += f", {total_errors} errors"
    summary += f"\nTotal crawl duration: {total_time:.1f}s ({total_time/60:.1f} minutes)"
    summary += "\n"

    return summary
