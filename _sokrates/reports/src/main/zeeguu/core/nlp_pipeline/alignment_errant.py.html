Work backwards from bottom right until we hit top left while i + j != 0: # Get the edit operation in the current cell op = self.op_matrix[i][j] # Matches and substitutions if op in {"M", "S"}: align_seq.append((op, i-1, i, j-1, j)) i -= 1 j -= 1 #
